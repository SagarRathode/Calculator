cmake_minimum_required(VERSION 3.10)

project(Calculator)

set(CMAKE_CXX_STANDARD 17)

add_executable(
    Calculator
    main.cpp
    Tokenizer.cpp
    Parser.cpp
)
//For compile
g++ -std=c++17 main.cpp Tokenizer.cpp Parser.cpp -o calculator

//clang++

clang++ -std=c++17 calculator.cpp Tokenizer.cpp Parser.cpp -o calculator
//Run

>./calculator

> 2 + 3 * 4
14


Design Advantages
This modular design gives you:
clean separation of responsibility,
reusable tokenizer,
extensible parser,
easier debugging,
scalable architecture,
professional C++ structure.
